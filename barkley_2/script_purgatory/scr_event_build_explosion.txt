/// scr_event_build_explosion(x,y,z,type)
/// This script is used to create an explosion at a specific point

var action = ds_list_create();
ds_list_add(action, scr_event_action_explosion);
ds_list_add(action, argument0); // x value 
ds_list_add(action, argument1); // y value
ds_list_add(action, argument2); // z value
ds_list_add(action, argument3); // type
ds_list_add(_event_list, action); // add the action to the list

return ds_list_size(_event_list) - 1;
