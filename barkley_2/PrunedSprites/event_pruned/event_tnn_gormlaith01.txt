// event_tnn_gormlaith01
// One of the rebel base cardplayers

if (argument0 == SCRIPT_START) {}
else if (argument0 == SCRIPT_STEP) {}
else if (argument0 == SCRIPT_INTERACT)
{
    // get rank
    var rank_gorm = scr_longinus_get_rank("gormlaith") + " Gormlaith";

    // Build Event
    var gormlaith = id;
    var event = scr_event_create_main();
    with (event) {
        // *** govQuest = 6 = Governor quest complete. ***
        if (scr_quest_get_state("govQuest") == 6) {
            if (scr_quest_get_state("longinusGov") >= 1) {
                scr_event_build_dialogue(rank_gorm, NULL, "Well, it looks like you can walk out the front gate now. As bad as it seems in Tir na Nog, it's a million times worse out there in the Necron 7 wilds. Take care of yourself.");
            }
            else {
                scr_event_build_dialogue(rank_gorm, NULL, "Well, you can go ahead and leave now. You know, you didn't do us any favors here in Tir na Nog, but it's still a million times worse out there in the Necron 7 wilds. I really don't mind that you're going!");
            }
        } 
        // *** govQuest = 5 = currently governor. ***
        else if (BodySwap() == "governor") {
            scr_event_build_dialogue(rank_gorm, NULL, "Not sure what smells worse, Duergar urine or Duergars.");
        }
        // *** govQuest = 4 = ready to become governor. ***
        else if (scr_quest_get_state("govQuest") == 4) {
            scr_event_build_dialogue(rank_gorm, NULL, "I joined L.O.N.G.I.N.U.S. because I don't think it's right that Cuchulainn can round up all the dwarfs, even if he thinks it's for our own good. I'm looking forward to the prize... but I'd rather be back in the Sombrero Galaxy.");
        }
        // *** govQuest = 3 = have duergar urine. ***
        else if (scr_quest_get_state("govQuest") == 3) {
            scr_event_build_dialogue(rank_gorm, NULL, "Uh, is that Duergar piss? You don't need to put it on the table.");
        }
        // *** govQuest = 2 = Governor quest accepted, need duergar urine. ***
        else if (scr_quest_get_state("govQuest") == 2) {
            scr_event_build_dialogue(rank_gorm, NULL, "Mr. Itkonen is the only one here who actually likes poker.");
        }
        // *** govQuest = 1 = Governor quest rejected ***
        else if (scr_quest_get_state("govQuest") == 1) {
            scr_event_build_dialogue(rank_gorm, NULL, "You know, on very rare occasions, the governor grants permission to leave to certain individuals, usually to take messages to the Duergar Mines. If you switched bodies with the governor, gave yourself permission to leave during his speech and then switched back, you could get away pretty easily. I mean uh, it's pretty far-fetched, but who knows...");
        }
        // *** govQuest = 0 ***
        else if (scr_quest_get_state("govQuest") == 0) {
            scr_event_build_dialogue(rank_gorm, NULL, "Wow, did you really make it all the way through the sewers just to get here? You must be pretty tough! I came here with a friend... but she didn't make it...");
        }
    }
    scr_event_advance(event);
}
