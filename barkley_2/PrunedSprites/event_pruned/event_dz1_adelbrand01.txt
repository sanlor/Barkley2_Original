// Event for Adelbrand, Mighty warrior, veteran, talks about the dangerous of Necron 7
//at time 1, 6, 7, 12

if (argument0 == SCRIPT_STEP) {
    if (scr_time_get() == 1) {
        scr_event_interactive_activate();
    }
    else if (scr_time_get() == 6) {
        scr_event_interactive_activate();
    }
    else if (scr_time_get() == 7) {
       scr_event_interactive_activate();
    }
    else if (scr_time_get() == 12) {
        scr_event_interactive_activate();
    }
    else {
        scr_event_interactive_deactivate();
    }
}
else if (argument0 == SCRIPT_INTERACT) {
    // Build Event
    var adelbrand = id;
    var event = scr_event_create_main();
    with (event) {
        
        //Time = 1
        if (scr_time_get() == 1) {
            if (scr_quest_get_state("adelbrandState") == 0) {
            //Never spoken to before.
                scr_event_build_dialogue("Adelbrand", NULL, "You look like you just got out of Tir na Nog.");
                scr_event_build_dialogue(P_NAME, s_port_hoopz, "Yeah, I did...");
                scr_event_build_dialogue("Adelbrand", NULL, "Then I suppose you're not familiar with the legend of the Knights of the Round Table, are you?");
                scr_event_build_dialogue(P_NAME, s_port_hoopz, "Not really...");
                scr_event_build_dialogue("Adelbrand", NULL, "Then I suppose you're not familiar with the legend of the Knights of the Round Table, are you?");
                scr_event_build_quest_state("adelbrandState", 1);
            } else {
                scr_event_build_dialogue("Adelbrand", NULL, "My name is Adelbrand and I'm a traveling warrior. Long ago there lived a conclave of 12 knights");
            }
        }
    }
    scr_event_advance(event);
}
