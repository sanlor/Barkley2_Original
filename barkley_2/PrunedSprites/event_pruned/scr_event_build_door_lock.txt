/// scr_event_build_door_lock(door name, lock text)
/// Builds an action to lock a door

var action = ds_list_create();
ds_list_add(action, scr_event_action_door_lock);
ds_list_add(action, argument0); // the door to lock
ds_list_add(action, argument1); // locked text
